sudo netplan apply	#使其生效

sudo vim /etc/default/isc-dhcp-server	#必须做的
sudo airbase-ng -c 6 -e "Free_wifi" wlan1mon	#生成一个伪装ap
sudo ifconfig at0 up	#打开at0端口
sudo service isc-dhcp-server restart	#启动at0这个端口的dhcp服务
sudo su		#切换到root
